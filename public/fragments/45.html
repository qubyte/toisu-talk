Toisu! server with async-await

async function checkAccess(req, res) {
  const authed = await getAccess(req);

  if (!authed) {
    res.statusCode = 401;
    res.end();
  }
}

const app = new Toisu()
  .use(checkAccess);
  .use((req, res) => sendPlain(req, res, 'Hello, world!'));

http.createServer(app.requestHandler).listen(3000);
