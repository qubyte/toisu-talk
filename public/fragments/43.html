Toisu! server

function checkAccess(req, res) {
  return getAccess(req).then(authed => {
    if (!authed) {
      res.status(401).end();
    }
  });
}

const app = new Toisu()
  .use(checkAccess);
  .use((req, res) => {
    res.writeHead(200, {'Content-Type': 'text/plain; charset=utf-8'});
    res.write('Hello, world!');
    res.end();
  });

http.createServer(app).listen(3000);
