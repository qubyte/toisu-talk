async function behaviour4() {
  try {
    const result = await somePromise;

    console.log('Success!', result);
  } catch (e) {
    console.error('Error!', e);
  }
}

// is the same as

function behaviour4() {
  return somePromise
    .then(result => console.log('Success!', result))
    .catch(e => console.error('Error!', e));
}
