function checkAccess(req, res, next) {
  getAccess(req, (err, authorized) => {
    if (err) {
      res.status(500).end();
    } else if (!authorized) {
      res.status(401).end();
    } else {
      next();
    }
  });
}

const app = express()
  .use(checkAccess);
  .use((req, res) => res.send('Hello, world!'));

http.createServer(app).listen(3000);
